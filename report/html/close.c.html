8 void processClose(ContractState* state) {
9 // Iterate over each internal account
10 for (int i = 0; i < state->internalWallet.numAccounts; i++) {
11 InternalAccount* account = &(state->internalWallet.accounts[i]);
12
13 // Find the corresponding party for the internal account
14 Party* externalParty = NULL;
15 for (int j = 0; j < state->numParties; j++) {
16 if (state->parties[j]->id == account->id) {
17 externalParty = state->parties[j];
18 break;
19 }
20 }
21
22 // Refund the tokens from the internal account to the external wallet
23 for (int j = 0; j < account->wallet.numTokens; j++) {
24 Token* token = &(account->wallet.tokens[j]);
25 for (int k = 0; k < externalParty->wallet.numTokens; k++) {
26 if (externalParty->wallet.tokens[k].currency == token->currency) {
27 externalParty->wallet.tokens[k].amount += token->amount;
28 break;
29 }
30 }
31 }
32
33 // Clear the internal account
34 for (int j = 0; j < account->wallet.numTokens; j++) {
35 account->wallet.tokens[j].amount = 0;
36 }
37 }
38 }
